/*
 Servo Motor Control using Potentiometer
 This project controls the angle of a servo motor using a potentiometer.
 Created as part of Arduino learning practice.
*/

#include <Servo.h>

Servo servoMotor;

const int POT_PIN = A0;   // Potentiometer connected to A0
const int SERVO_PIN = 9;  // Servo signal pin

int potValue = 0;
int servoAngle = 0;

void setup() {
  pinMode(POT_PIN, INPUT);
  servoMotor.attach(SERVO_PIN);
}

void loop() {
  // Read potentiometer value
  potValue = analogRead(POT_PIN);

  // Map potentiometer value (0–1023) to servo angle (0–180)
  servoAngle = map(potValue, 0, 1023, 0, 180);

  // Rotate servo motor
  servoMotor.write(servoAngle);

  delay(15);  // Small delay for smooth movement
}
